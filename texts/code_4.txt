from selenium import webdriver
from time import sleep

chrome = webdriver.Chrome()
chrome.get('https://www.bastter.com')
sleep(1)
chrome.maximize_window()
chrome.find_element_by_link_text('Fazer Login').click()
sleep(1)
chrome.find_element_by_id('login-email').send_keys('<login>')
chrome.find_element_by_id('login-password').send_keys('<senha>')
